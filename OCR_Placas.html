import cv2 import pytesseract # CAMBIA ESTA RUTA A DONDE INSTALASTE TESSERACT EN TU COMPUTADORA pytesseract.pytesseract.tesseract_cmd = r"C:\Users\MATUTINO\AppData\Local\Programs\Tesseract-OCR\tesseract.exe" # Lista de placas autorizadas en el estacionamiento del CBTIS144 placas_autorizadas = ['XYZ789', 'QWEYQ87', 'NITGUILTY', '["BL445567|'] print("="*60) print("SISTEMA DE CONTROL DE ACCESO - ESTACIONAMIENTO CBTIS 144") print("="*60) # 1. CARGAR 2 IMAGENES DE PLACAS print("\n--- CARGANDO IMAGENES ---\n") # Primera placa placa1 = cv2.imread("placas3.jpg") placanueva1 = cv2.resize(placa1, (800, 300)) # Segunda placa placa2 = cv2.imread("placas4.png") placanueva2 = cv2.resize(placa2, (800, 300)) # 2. EXTRAER EL TEXTO DE LAS 2 PLACAS print("--- EXTRAYENDO TEXTO ---\n") # Procesar primera placa placa_gris1 = cv2.cvtColor(placanueva1, cv2.COLOR_BGR2GRAY) placa_res1 = cv2.threshold(placa_gris1, 120, 255, cv2.THRESH_BINARY)[1] texto_placa1 = pytesseract.image_to_string(placa_res1, lang='eng') texto_limpio1 = texto_placa1.strip().replace(" ", "").replace("-", "").upper() print("Placa 1 detectada: " + texto_limpio1) # Procesar segunda placa placa_gris2 = cv2.cvtColor(placanueva2, cv2.COLOR_BGR2GRAY) placa_res2 = cv2.threshold(placa_gris2, 120, 255, cv2.THRESH_BINARY)[1] texto_placa2 = pytesseract.image_to_string(placa_res2, lang='eng') texto_limpio2 = texto_placa2.strip().replace(" ", "").replace("-", "").upper() print("Placa 2 detectada: " + texto_limpio2) # 3. RECORRER LA LISTA DE LAS PLACAS PERMITIDAS print("\n--- LISTA DE PLACAS AUTORIZADAS ---") for placa in placas_autorizadas: print(" * " + placa) # 4. MOSTRAR UN MENSAJE DE LAS PLACAS RESTANTES (LAS QUE NO SE ESCANEARON) print("\n--- PLACAS NO ESCANEADAS (RESTANTES) ---") placas_escaneadas = [] placas_escaneadas.append(texto_limpio1) placas_escaneadas.append(texto_limpio2) print("Las siguientes placas autorizadas NO fueron escaneadas:") for placa in placas_autorizadas: if placa not in placas_escaneadas: print(" * " + placa) # 6.
